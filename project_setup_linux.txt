# Required Softwares
# 1. XXAMP
#    1. Open XXAMP
#    2. Start Apache & MySql
#       Click on MySql->Admin
#    3. Create Database

# 2. Python 3.8.0 or newer
# Installing VirtualEnvironment
# 1. Navigate to your project directory
cd /path/to/project

# 2. Install virtualenv
pip install virtualenv

# Naming VirtualEnvironment
# 3. Create a virtual environment named "venv"
virtualenv venv

# Activating VirtualEnvironment
# 4. Activate the virtual environment
source venv/bin/activate

# Installing Django and other dependencies
# 5. Install Django and required packages
pip install Django
pip install requests
pip install beautifulsoup4
pip install mysqlclient

# Starting project
# 6. Create a Django project named "projectname"
django-admin startproject projectname

# Creating App
# 7. Navigate to the project directory
cd projectname

# 8. Create a Django app named "appname"
python manage.py startapp appname

# Running Project
# 9. Run the Django development server
python manage.py runserver
